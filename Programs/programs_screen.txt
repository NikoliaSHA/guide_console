# =================== screen {http://xgu.ru/wiki/Man:screen} ===================

screen -x                       # подключится к активной сессии
screen -ls | -list              # список сессий
screen -r 22545                 # восстановление сессии по номеру сессии
screen -c                       # подключить файл конф-ций
Ctrl+a : source ~/.screenrc     # подключить файл конФ-ций
Ctrl+a "                        # показать список вкладок
Ctrl+a a                        # в начало строки
Ctrl+a c                        # создать новую вкладку 
Ctrl+a K | Ctrl + d             # удалить текущую вкладку
Ctrl+a C                        # очистить текущую вкладку
Ctrl+a A                        # переименновать вкладку
Ctrl+a m                        # repeat lost message
Ctrl+a space | n                # показать следующую вкладку
Ctrl+a backspace | h | p        # показать предыдущую вкладку
Ctrl+s | Ctrl+q                 # xon/xoff

Ctrl+a tab                      # Переключить фокус ввода на следующий регион.
Ctrl+a Q                        # Удалить все регионы за исключением текущего

Ctrl+a d                        # выйти из screen
