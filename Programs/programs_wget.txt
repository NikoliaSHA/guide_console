# =============================== wGet =========================================

wget http://file.zip            # скачивание файла file.zip в текущую директорию
wget -P /path http://file.zip   # скачивание в директорию /path/to/save
wget -c http://file.zip         # докачивание файла file.zip в случаи обрыва
wget -O ar.zip http://file.zip  # скачивание файла file.zip и сохранение под именем arch.zip
wget -i files.txt               # скачивание файлов из списка в files.txt
wget --tries=10 http://file.zip # количество попыток на скачивание
wget -Q5m -i http://site        # квота на макс. размер скачанных файлов (-r)
echo 'wget http://file.zip' | at 12:00 # скачать файл в 12:00
wget ftp://*.zip    			# скачивание всех файлов по шаблону
wget http://file{1..10}.zip 	# скачивание всех файлов по шаблону
wget -S http://example.com/ 	# вывод заголовков HTTP серверов
wget --spider -i urls.txt   	# проверка ссылок в файле на доступность
wget -b http://file.zip 		# скачивание в фоне, лог wget.log
wget --limit-rate=20k http:// 	# ограничение скорости скачивания
wget -R bmp http://				# не скачивать bmp файлы
wget -A png,jpg http://			# скачивать только файлы png и jpg

-r 								# ходим по ссылкам (рекурсивно)
-k 								# преобразовываем ссылки
-p 								# скачивание для html-стр. (стили,картинки)
-l 								# глубина скачивания, 0 - бесконечная влож.
-nc 							# не перезаписывать существующие файлы
-np 							# не подниматься выше начального адреса

Часто используемые wget параметры можно вынести в ~/.wgetrc
