# =================== Выражения для проверки файлов ============================

file1 -ef file2                 # file1 и file2 имеют одно и то же число inode
file1 -nt file2                 # file1 новее fileа file2
file1 -ot file2                 # file1 старше fileа file2
[ ext1 -a ext2 ]                # и аналог &&
-b file                         # file существует и является блочным
-с file                         # file существует и является символьным
-d file                         # file существует и является каталогом
-e file                         # file существует
-f file                         # file существует и является обычным fileом
-g file                         # file существует и имеет атрибут set-group-ID
-G file                         # file существует и принадлежит действующей группе
-k file                         # file существует и имеет атрибут «sticky bit»
-L file                         # file существует и является символической ссылкой
[ ext1 -o ext2 ]                # или аналог ||
-O file                         # file существует и принадлежит дей-му пользователю
-p file                         # file существует и является именованным каналом
-r file                         # file существует и доступен для чтения 
-s file                         # file существует и имеет размер больше нуля
-S file                         # file существует и является сетевым сокетом
-t дескриптор_файла             # для проверки стандартных ввода/вывода/ошибок
-u file                         # file существует и имеет атрибут setuid
-w file                         # file существует и доступен для записи
-x file                         # file существует и доступен для выполнения

# ====================== Выражения для проверки строк ==========================

str                             # Cтрока не пустая
-n str                          # Длина строки больше нуля
-z str                          # Длина строки равна нулю
str1 = str2                     # str1 и str2 равны, можно один =
str1 == str2                    # str1 и str2 равны, предпочтительнее два ==
str1 != str2                    # str1 и str2 не равны
str1 > str2                     # str1 больше, чем str2, по алфавиту
str1 < str2                     # str1 меньше, чем str2, по алфавиту
str =~ рег_выражение            # [[]] составная команда test


# ==================== Выражения для проверки целых чисел ======================

число1 -eq число2               # число1 и число2 равны
число1 -ne число2               # число1 и число2 не равны
число1 -le число2               # число1 меньше или равно числу2
число1 -lt число2               # число1 меньше, чем число2
число1 -ge число2               # число1 больше или равно числу2
число1 -gt число2               # число1 больше, чем число2
